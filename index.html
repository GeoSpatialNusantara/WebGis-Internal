let activeLayers = {}; let folderChildren = {}; function createCheckbox(name, level=0){ let div = document.createElement("div"); div.style.marginLeft = (level * 15) + "px"; let checkbox = document.createElement("input"); checkbox.type = "checkbox"; div.appendChild(checkbox); div.appendChild(document.createTextNode(" " + name)); return {div, checkbox}; } async function loadFolder(path, container, level=0){ const response = await fetch(path); const items = await response.json(); for(const item of items){ // ===================== // FOLDER // ===================== if(item.type === "dir"){ let {div, checkbox} = createCheckbox(item.name, level); container.appendChild(div); let subContainer = document.createElement("div"); container.appendChild(subContainer); folderChildren[item.name] = []; checkbox.onchange = function(){ subContainer.querySelectorAll("input[type=checkbox]") .forEach(cb => cb.checked = checkbox.checked); subContainer.querySelectorAll("input[type=checkbox]") .forEach(cb => cb.dispatchEvent(new Event("change"))); }; await loadFolder(item.url, subContainer, level+1); } // ===================== // FILE GEOJSON // ===================== if(item.type === "file" && item.name.endsWith(".geojson")){ let rawURL = `https://raw.githubusercontent.com/${owner}/${repo}/${branch}/${item.path}`; let {div, checkbox} = createCheckbox(item.name, level); container.appendChild(div); checkbox.onchange = function(){ if(this.checked){ fetch(rawURL) .then(r=>r.json()) .then(data=>{ let geo = L.geoJSON(data,{ onEachFeature: function(feature, layer){ if(feature.properties){ layer.bindPopup( Object.entries(feature.properties) .map(([k,v]) => k + ": " + v) .join("
") ); } } }).addTo(map); activeLayers[item.path] = geo; map.fitBounds(geo.getBounds()); }); }else{ if(activeLayers[item.path]){ map.removeLayer(activeLayers[item.path]); delete activeLayers[item.path]; } } }; } } }
